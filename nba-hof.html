NBA Hall of Fame Prediction — October 2025

End-to-end classification on an imbalanced dataset; focus on PR-AUC, F1 for HoF class, and threshold tuning.

Problem

Predict whether an NBA player will be inducted into the Hall of Fame from career statistics. The dataset is highly imbalanced (~3% HoF), so accuracy alone is misleading. I optimize for the minority class (HoF) using PR-AUC and F1, and compare multiple approaches: class weighting, SMOTE, feature engineering, threshold tuning, and tree-based models.

Highlights

  • Top model: Gradient Boosting + Feature Engineering, tuned threshold ≈ 0.53.
  • Balanced operating point: Precision ≈ Recall ≈ F1 ≈ 0.72 for the HoF class at 98.3% overall accuracy.
  • PR-AUC ≈ 0.72 (vs logistic regression ≈ 0.65–0.70), confirming robust ranking under class imbalance.
  • Logistic Regression baseline (scaled, 0.5 threshold): Prec 0.76, Rec 0.66, F1 0.70; improved via SMOTE and threshold tuning.

# Creating a function to parse various birthday formats
def parse_birthday(x):
    if pd.isna(x) or str(x).strip().lower() == "unknown":
        return pd.NaT

    x = str(x).strip()

    # Case 1: looks like full timestamp "YYYY-MM-DD 00:00:00"
    try:
        return pd.to_datetime(x, format="%Y-%m-%d %H:%M:%S", errors="raise")
    except Exception:
        pass

    # Case 2: just a 4-digit year
    if x.isdigit() and len(x) == 4:
        return pd.to_datetime(f"{x}-01-01", errors="coerce")

    # Case 3: month/day/year
    try:
        return pd.to_datetime(x, format="%m/%d/%Y", errors="coerce")
    except Exception:
        # Fallback to pandas' general parser
        return pd.to_datetime(x, errors="coerce")

class EvalRow:
    model: str

def summarize(y_true, y_pred, y_score, name, variant="", threshold=np.nan):
    acc = accuracy_score(y_true, y_pred)
    # metrics for positive class (label=1)
    prec, rec, f1, sup = precision_recall_fscore_support(y_true, y_pred, average=None, labels=[1,0])
    pr_auc = average_precision_score(y_true, y_score) if y_score is not None else np.nan
    roc = roc_auc_score(y_true, y_score) if y_score is not None else np.nan
    return EvalRow(name, variant, threshold, acc, roc, pr_auc, prec[0], rec[0], f1[0], sup[0])

def find_best_f1_threshold(y_true, y_score):
    p, r, thr = precision_recall_curve(y_true, y_score)
    f1 = 2 * (p[:-1]*r[:-1]) / (p[:-1] + r[:-1] + 1e-12)
    i = np.argmax(f1)
    return float(thr[i]), float(f1[i]), float(p[i]), float(r[i])

def find_threshold_for_recall(y_true, y_score, target_recall=0.80):
    p, r, thr = precision_recall_curve(y_true, y_score)
    # find first threshold whose recall >= target
    idx = np.where(r[:-1] >= target_recall)[0]
    if len(idx) == 0:
        return None  # not achievable
    i = idx[-1]  # highest precision among recalls >= target tends to be at larger thresholds -> use last
    return float(thr[i]), float(p[i]), float(r[i])

def show_curves(y_true, y_score, title):
    # ROC
    fpr, tpr, _ = roc_curve(y_true, y_score)
    plt.figure()
    plt.plot(fpr, tpr, label=title)
    plt.plot([0,1],[0,1],'--')

X_fe['pts_per_g'] = X_fe['career_pts'] / (X_fe['career_g'] + eps)

# Signed log that handles negatives: slog1p(x) = sign(x) * log1p(|x|)

    x = np.asarray(x, dtype=float)
    return np.sign(x) * np.log1p(np.abs(x))

X_fe['log_career_ws']  = slog1p(X_fe['career_ws'])   # <-- handles negatives safely
